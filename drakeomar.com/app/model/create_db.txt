CREATE TABLE static_dim (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  screenWidth MEDIUMINT UNSIGNED NOT NULL,
  screenHeight MEDIUMINT UNSIGNED NOT NULL,
  windowWidth MEDIUMINT UNSIGNED NOT NULL,
  windowHeight MEDIUMINT UNSIGNED NOT NULL,
  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE static_allowed (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  acceptCookies BOOLEAN NOT NULL,
  allowJS BOOLEAN NULL,
  allowCSS BOOLEAN NULL,
  allowImages BOOLEAN NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE static_gen (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  userAgent VARCHAR(255) NULL,
  lang VARCHAR(255) NULL,
  connType VARCHAR(25) NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE performance (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  perf JSON NOT NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE performance_times (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  startTime VARCHAR(24) NOT NULL,
  endTime VARCHAR(24) NULL,
  totalTime MEDIUMINT UNSIGNED NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);


CREATE TABLE activity_keys (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  keydown VARCHAR(10) NOT NULL,
  timestamp VARCHAR(255) NOT NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE activity_page (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  enteredPage VARCHAR(255) NOT NULL,
  leftPage VARCHAR(255) NULL,
  whichPage VARCHAR(255) NOT NULL

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE activity_click (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  click_val JSON NULL,
  timestamp VARCHAR(255) NOT NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE activity_mouse (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  cursor_val JSON NULL,
  scroll_val JSON NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

  FOREIGN KEY(usernum) REFERENCES users(id)

CREATE TABLE check_css (

  allowed BOOLEAN NOT NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)

CREATE TABLE check_js (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,

CREATE TABLE activity_mouse (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  cursor_val JSON NULL,
  click JSON NULL,
  scroll JSON NULL,

  PRIMARY KEY(id)
);

CREATE TABLE activity_idle (
  id MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
  usernum INT UNSIGNED NOT NULL,
  breakStarted VARCHAR(25) NOT NULL,
  breakEnded VARCHAR(25) NOT NULL,
  idleTime INT UNSIGNED NOT NULL,

  PRIMARY KEY(id),
  FOREIGN KEY(usernum) REFERENCES users(id)
);

CREATE TABLE users (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  sessionId VARCHAR(24) NOT NULL UNIQUE,

  PRIMARY KEY(id)
);

CREATE TABLE login_users (
  id INT UNSIGNED NOT NULL AUTO_INCREMENT,
  username VARCHAR(24) NOT NULL, 
  email VARCHAR(64) NOT NULL, 
  password VARCHAR(255) NOT NULL, 
  token VARCHAR(255) NOT NULL,
  PRIMARY KEY(id)
);
